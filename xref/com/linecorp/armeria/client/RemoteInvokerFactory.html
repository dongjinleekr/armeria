49  /**
50   * Creates and manages {@link RemoteInvoker}s.
51   *
52   * <h3>Life cycle of the default {@link RemoteInvokerFactory}</h3>
53   * <p>
54   * {@link Clients} or {@link ClientBuilder} uses {@link #DEFAULT}, the default {@link RemoteInvokerFactory},
55   * unless you specified a {@link RemoteInvokerFactory} explicitly. Calling {@link #close()} on the default
56   * {@link RemoteInvokerFactory} won't terminate its I/O threads and release other related resources unlike
57   * other {@link RemoteInvokerFactory} to protect itself from accidental premature termination.
58   * </p><p>
59   * Instead, when the current {@link ClassLoader} is {@linkplain ClassLoader#getSystemClassLoader() the system
60   * class loader}, a {@link Runtime#addShutdownHook(Thread) shutdown hook} is registered so that they are
61   * released when the JVM exits.
62   * </p><p>
63   * If you are in an environment managed by a container or you desire the early termination of the default
64   * {@link RemoteInvokerFactory}, use {@link #closeDefault()}.
65   * </p>
66   */
67  public final class RemoteInvokerFactory implements AutoCloseable {
68  
69      private static final Logger logger = LoggerFactory.getLogger(RemoteInvokerFactory.class);
70  
71      private enum TransportType {
72          NIO, EPOLL
73      }
74  
75      /**
76       * The default {@link RemoteInvokerFactory} implementation.
77       */
78      public static final RemoteInvokerFactory DEFAULT = new RemoteInvokerFactory(
79              RemoteInvokerOptions.DEFAULT,
80              type -> {
81                  switch (type) {
82                      case NIO:
83                          return new DefaultThreadFactory("default-armeria-client-nio", true);
84                      case EPOLL:
85                          return new DefaultThreadFactory("default-armeria-client-epoll", true);
86                      default:
87                          throw new Error();
88                  }
89              });
90  
91      /**
92       * Closes the default {@link RemoteInvokerFactory}.
93       */
94      public static void closeDefault() {
95          logger.debug("Closing the default {}", RemoteInvokerFactory.class.getSimpleName());
96          DEFAULT.close0();
97      }
98  
99      static {
100         if (RemoteInvokerFactory.class.getClassLoader() == ClassLoader.getSystemClassLoader()) {
101             Runtime.getRuntime().addShutdownHook(new Thread(RemoteInvokerFactory::closeDefault));
102         }
103     }
104 
105     private static final ThreadFactory DEFAULT_THREAD_FACTORY_NIO =
106             new DefaultThreadFactory("armeria-client-nio", false);
107 
108     private static final ThreadFactory DEFAULT_THREAD_FACTORY_EPOLL =
109             new DefaultThreadFactory("armeria-client-epoll", false);
110 
111     private final EventLoopGroup eventLoopGroup;
112     private final boolean closeEventLoopGroup;
113     private final Map<SessionProtocol, RemoteInvoker> remoteInvokers;
114 
115     /**
116      * Creates a new instance with the specified {@link RemoteInvokerOptions}.
117      */
118     public RemoteInvokerFactory(RemoteInvokerOptions options) {
119         this(options, type -> {
120             switch (type) {
121                 case NIO:
122                     return DEFAULT_THREAD_FACTORY_NIO;
123                 case EPOLL:
124                     return DEFAULT_THREAD_FACTORY_EPOLL;
125                 default:
126                     throw new Error();
127             }
128         });
129     }
130 
131     private RemoteInvokerFactory(RemoteInvokerOptions options,
132                                  Function<TransportType, ThreadFactory> threadFactoryFactory) {
133 
134         requireNonNull(options, "options");
135         requireNonNull(threadFactoryFactory, "threadFactoryFactory");
136 
137         final Bootstrap baseBootstrap = new Bootstrap();
138 
139         baseBootstrap.channel(channelType());
140         baseBootstrap.resolver(
141                 options.addressResolverGroup()
142                        .orElseGet(() -> new DnsAddressResolverGroup(
143                                datagramChannelType(), DnsServerAddresses.defaultAddresses())));
144 
145         baseBootstrap.option(ChannelOption.CONNECT_TIMEOUT_MILLIS,
146                              ConvertUtils.safeLongToInt(options.connectTimeoutMillis()));
147         baseBootstrap.option(ChannelOption.SO_KEEPALIVE, true);
148 
149         final Optional<EventLoopGroup> eventLoopOption = options.eventLoopGroup();
150         if (eventLoopOption.isPresent()) {
151             eventLoopGroup = eventLoopOption.get();
152             closeEventLoopGroup = false;
153         } else {
154             eventLoopGroup = createGroup(threadFactoryFactory);
155             closeEventLoopGroup = true;
156         }
157 
158         final EnumMap<SessionProtocol, RemoteInvoker> remoteInvokers = new EnumMap<>(SessionProtocol.class);
159         final HttpRemoteInvoker remoteInvoker = new HttpRemoteInvoker(baseBootstrap, options);
160 
161         SessionProtocol.ofHttp().forEach(
162                 protocol -> remoteInvokers.put(protocol, remoteInvoker));
163 
164         this.remoteInvokers = Collections.unmodifiableMap(remoteInvokers);
165     }
166 
167     private static Class<? extends SocketChannel> channelType() {
168         return NativeLibraries.isEpollAvailable() ? EpollSocketChannel.class : NioSocketChannel.class;
169     }
170 
171     private static Class<? extends DatagramChannel> datagramChannelType() {
172         return NativeLibraries.isEpollAvailable() ? EpollDatagramChannel.class : NioDatagramChannel.class;
173     }
174 
175     private static EventLoopGroup createGroup(Function<TransportType, ThreadFactory> threadFactoryFactory) {
176         return NativeLibraries.isEpollAvailable() ?
177                new EpollEventLoopGroup(0, threadFactoryFactory.apply(TransportType.EPOLL)) :
178                new NioEventLoopGroup(0, threadFactoryFactory.apply(TransportType.NIO));
179     }
180 
181     /**
182      * Returns the {@link EventLoopGroup} being used by this remote invoker factory. Can be used to, e.g.,
183      * schedule a periodic task without creating a separate event loop.
184      */
185     public EventLoopGroup eventLoopGroup() {
186         return eventLoopGroup;
187     }
188 
189     /**
190      * Returns a {@link RemoteInvoker} that can handle the specified {@link SessionProtocol}.
191      */
192     public RemoteInvoker getInvoker(SessionProtocol sessionProtocol) {
193         final RemoteInvoker remoteInvoker = remoteInvokers.get(sessionProtocol);
194         if (remoteInvoker == null) {
195             throw new IllegalArgumentException("unsupported session protocol: " + sessionProtocol);
196         }
197         return remoteInvoker;
198     }
199 
200     /**
201      * Closes all {@link RemoteInvoker}s managed by this factory and shuts down the {@link EventLoopGroup}
202      * created implicitly by this factory.
203      */
204     @Override
205     public void close() {
206         // The global default should never be closed.
207         if (this == DEFAULT) {
208             logger.debug("Refusing to close the default {}; must be closed via closeDefault()",
209                          RemoteInvokerFactory.class.getSimpleName());
210             return;
211         }
212 
213         close0();
214     }
215 
216     private void close0() {
217         remoteInvokers.forEach((k, v) -> v.close());
218         if (closeEventLoopGroup) {
219             eventLoopGroup.shutdownGracefully().syncUninterruptibly();
220         }
221     }
222 }
